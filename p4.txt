import pretty_midi, numpy as np

midi = pretty_midi.PrettyMIDI()

# Instruments: Piano, Violin, Flute
insts = [pretty_midi.Instrument(program=p) for p in [0, 40, 73]]

for idx, inst in enumerate(insts):
    for i in range(10):
        pitch = np.random.randint(60, 72) + idx*3
        start = i * 0.5
        note = pretty_midi.Note(velocity=90, pitch=pitch, start=start, end=start+0.5)
        inst.notes.append(note)
    midi.instruments.append(inst)

midi.write("music.mid")
print("Music generated: music.mid")
