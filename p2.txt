from transformers import MarianMTModel, MarianTokenizer
import torch

# Language models
models = {
    "Hindi": "Helsinki-NLP/opus-mt-en-hi",
    "Tamil": "Helsinki-NLP/opus-mt-en-ta",
    "French": "Helsinki-NLP/opus-mt-en-fr",
    "German": "Helsinki-NLP/opus-mt-en-de"
}

device = "cuda" if torch.cuda.is_available() else "cpu"

print("Languages:", ", ".join(models.keys()))

while True:
    text = input("Enter text: ")
    if text.lower() == "exit":
        break

    lang = input("Translate to: ")
    if lang not in models:
        print("Not supported\n")
        continue

    tok = MarianTokenizer.from_pretrained(models[lang])
    model = MarianMTModel.from_pretrained(models[lang]).to(device)

    inp = tok([text], return_tensors="pt", padding=True).to(device)
    out = model.generate(**inp)

    print("Translated:", tok.decode(out[0], skip_special_tokens=True), "\n")
