from transformers import AutoTokenizer, AutoModelForCausalLM
import torch

# Load model
name = "bigcode/starcoder2-3b"
tok = AutoTokenizer.from_pretrained(name)
model = AutoModelForCausalLM.from_pretrained(
    name, torch_dtype=torch.float16, device_map="auto"
)

# User input
task = input("Enter programming task: ")

prompt = f"Write only code for: {task}\n"

inp = tok(prompt, return_tensors="pt").to(model.device)
out = model.generate(
    **inp,
    max_new_tokens=150,
    temperature=0.1,
    do_sample=False,
    pad_token_id=tok.eos_token_id
)

res = tok.decode(out[0], skip_special_tokens=True)

print("\nGenerated Code:\n")
print(res.replace(prompt, "").strip())
